# Redux store에 state 보관하는 법
만들어준 작명.js 파일을 열어서 아래와 같이 코드를 짜면 state 만들 수 있음

        import { configureStore, createSlice } from '@reduxjs/toolkit'

        let user = createSlice({
        name : 'user',
        initialState : 'kim'
        })

        export default configureStore({
        reducer: {
            user : user.reducer
        }
        })
